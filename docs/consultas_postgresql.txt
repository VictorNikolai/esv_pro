-- Ver todas las órdenes de trabajo (más recientes primero)
SELECT
    o.order_no,
    o.status,
    c.empresa,
    c.ruc,
    u.nombre    AS responsable,
    COUNT(oi.id) AS n_instrumentos,
    o.total_estimated AS total_s,
    DATE_FORMAT(o.created_at, '%d/%m/%Y') AS fecha
FROM work_orders o
JOIN clients c ON c.id = o.client_id
JOIN users   u ON u.id = o.created_by
LEFT JOIN order_instruments oi ON oi.order_id = o.id
GROUP BY o.id
ORDER BY o.created_at DESC;

-- Órdenes por estado
SELECT order_no, status, total_estimated,
       DATE_FORMAT(created_at, '%d/%m/%Y') AS fecha
FROM work_orders
WHERE status = 'borrador'    -- cambiar estado según necesidad
                              -- 'borrador' | 'ingresado' | 'cotizado'
                              -- 'aprobado' | 'en_proceso' | 'finalizado'
ORDER BY created_at DESC;

-- Órdenes de un usuario específico
SELECT o.order_no, o.status, c.empresa, o.total_estimated,
       DATE_FORMAT(o.created_at, '%d/%m/%Y') AS fecha
FROM work_orders o
JOIN clients c ON c.id = o.client_id
JOIN users   u ON u.id = o.created_by
WHERE u.username = 'm.rios'   -- reemplazar con el username
ORDER BY o.created_at DESC;

-- Órdenes por cliente (empresa)
SELECT o.order_no, o.status, o.total_estimated,
       DATE_FORMAT(o.created_at, '%d/%m/%Y') AS fecha
FROM work_orders o
JOIN clients c ON c.id = o.client_id
WHERE c.empresa LIKE '%Energy%'   -- reemplazar con nombre del cliente
ORDER BY o.created_at DESC;

-- Detalle de una orden específica
SELECT
    o.order_no, o.status, o.total_estimated,
    c.empresa, c.ruc, c.contacto, c.telefono, c.email,
    cm.name AS metodo, oi.marca, oi.serie, oi.alcance,
    oi.subtotal,
    GROUP_CONCAT(ip.point_label SEPARATOR ' | ') AS puntos
FROM work_orders o
JOIN clients           c  ON c.id  = o.client_id
JOIN order_instruments oi ON oi.order_id = o.id
JOIN catalog_methods   cm ON cm.id = oi.method_id
LEFT JOIN instrument_points ip ON ip.instrument_id = oi.id
WHERE o.order_no = 'ESV-OT-2026-0001'   -- reemplazar con el N° de orden
GROUP BY oi.id;

-- Cambiar estado de una orden
UPDATE work_orders
SET status = 'finalizado'              -- nuevo estado
WHERE order_no = 'ESV-OT-2026-0001';  -- reemplazar con el N° de orden

-- Órdenes del mes actual
SELECT o.order_no, c.empresa, o.status, o.total_estimated
FROM work_orders o
JOIN clients c ON c.id = o.client_id
WHERE YEAR(o.created_at)  = YEAR(NOW())
  AND MONTH(o.created_at) = MONTH(NOW())
ORDER BY o.created_at DESC;

-- Resumen de ingresos por mes (año actual)
SELECT
    MONTHNAME(created_at) AS mes,
    MONTH(created_at)     AS n_mes,
    COUNT(*)              AS total_ordenes,
    SUM(total_estimated)  AS ingresos_estimados
FROM work_orders
WHERE YEAR(created_at) = YEAR(NOW())
  AND status NOT IN ('borrador')
GROUP BY MONTH(created_at), MONTHNAME(created_at)
ORDER BY n_mes;

-- Ingresos totales por año
SELECT
    YEAR(created_at)     AS anio,
    COUNT(*)             AS ordenes,
    SUM(total_estimated) AS total_estimado
FROM work_orders
WHERE status IN ('finalizado', 'en_proceso', 'aprobado')
GROUP BY YEAR(created_at)
ORDER BY anio DESC;

-- Ver catálogo completo con puntos
SELECT
    cm.id, cm.code, cm.area, cm.name,
    cm.tariff AS tarifa_por_punto,

    COUNT(cp.id) AS n_puntos,
    cm.active

GROUP BY cm.id
ORDER BY cm.area, cm.name;

-- Ver métodos por área
SELECT name, tariff, procedure_code, procedure_description
FROM catalog_methods
WHERE area = 'Química'   -- cambiar: 'Dimensional' | 'Mecánica' | etc.
  AND active = 1;

-- Ver los puntos de calibración de un método
SELECT cm.name AS metodo, cp.point_label AS punto,
       cp.uncertainty AS incertidumbre, cp.sort_order
FROM calibration_points cp
JOIN catalog_methods cm ON cm.id = cp.method_id
WHERE cm.code = 'cf-matraz-aforo'   -- reemplazar con el código del método
ORDER BY cp.sort_order;

-- Métodos más utilizados en órdenes
SELECT cm.name, cm.area, COUNT(oi.id) AS veces_usado
FROM catalog_methods cm
JOIN order_instruments oi ON oi.method_id = cm.id
GROUP BY cm.id
ORDER BY veces_usado DESC
LIMIT 10;

-- Ver todos los métodos con su descripción de procedimiento
SELECT code, name, procedure_code, procedure_description
FROM catalog_methods
WHERE active = 1
ORDER BY area, name;

  SECCIÓN 6 — DOCUMENTOS GENERADOS

-- Ver documentos generados (más recientes primero)
SELECT
    d.doc_type AS tipo, d.doc_number AS numero,
    o.order_no AS orden,
    c.empresa  AS cliente,
    u.nombre   AS generado_por,
    DATE_FORMAT(d.generated_at, '%d/%m/%Y %H:%i') AS fecha
FROM documents d
JOIN work_orders o ON o.id = d.order_id
JOIN clients     c ON c.id = o.client_id
JOIN users       u ON u.id = d.generated_by
ORDER BY d.generated_at DESC
LIMIT 50;

-- Ver constancias generadas este mes
SELECT d.doc_number, o.order_no, c.empresa,
       DATE_FORMAT(d.generated_at, '%d/%m/%Y %H:%i') AS fecha
FROM documents d
JOIN work_orders o ON o.id = d.order_id
JOIN clients     c ON c.id = o.client_id
WHERE d.doc_type = 'constancia'
  AND MONTH(d.generated_at) = MONTH(NOW())
  AND YEAR(d.generated_at)  = YEAR(NOW())
ORDER BY d.generated_at DESC;

-- Ver cotizaciones generadas
SELECT d.doc_number, o.order_no, c.empresa, o.total_estimated,
       DATE_FORMAT(d.generated_at, '%d/%m/%Y %H:%i') AS fecha
FROM documents d
JOIN work_orders o ON o.id = d.order_id
JOIN clients     c ON c.id = o.client_id
WHERE d.doc_type = 'cotizacion'
ORDER BY d.generated_at DESC;

-- Ver notificaciones no leídas de todos los usuarios
SELECT
    u.nombre AS usuario, n.tipo, n.titulo,
    n.mensaje, n.referencia,
    DATE_FORMAT(n.created_at, '%d/%m/%Y %H:%i') AS fecha
FROM notifications n
JOIN users u ON u.id = n.user_id
WHERE n.leida = 0
ORDER BY n.created_at DESC;

-- Ver notificaciones de un usuario
SELECT tipo, titulo, mensaje, referencia,
       leida, DATE_FORMAT(created_at, '%d/%m/%Y %H:%i') AS fecha
FROM notifications
WHERE user_id = 1   -- reemplazar con el ID del usuario
ORDER BY leida ASC, created_at DESC;

-- Marcar todas las notificaciones de un usuario como leídas
UPDATE notifications SET leida = 1
WHERE user_id = 1;   -- reemplazar con el ID del usuario

-- Eliminar notificaciones antiguas (más de 90 días)
DELETE FROM notifications
WHERE created_at < DATE_SUB(NOW(), INTERVAL 90 DAY);

-- Contar notificaciones no leídas por usuario
SELECT u.nombre, u.role, COUNT(n.id) AS no_leidas
FROM users u
LEFT JOIN notifications n ON n.user_id = u.id AND n.leida = 0
WHERE u.active = 1
GROUP BY u.id
ORDER BY no_leidas DESC;

-- Ver solicitudes de registro pendientes
SELECT id, nombre_completo, username, email,
       cargo, empresa, rol_solicitado,
       DATE_FORMAT(created_at, '%d/%m/%Y %H:%i') AS fecha
FROM user_requests
WHERE estado = 'pendiente'
ORDER BY created_at ASC;

-- Ver todos los registros (pendientes, aprobados, rechazados)
SELECT
    r.nombre_completo, r.username, r.email,
    r.rol_solicitado, r.estado,
    u.nombre AS aprobado_por,
    DATE_FORMAT(r.created_at, '%d/%m/%Y') AS solicitud,
    DATE_FORMAT(r.revisado_at, '%d/%m/%Y') AS revision
FROM user_requests r
LEFT JOIN users u ON u.id = r.revisado_por
ORDER BY FIELD(r.estado,'pendiente','aprobada','rechazada'), r.created_at DESC;

  SECCIÓN 9 — REPORTES Y ESTADÍSTICAS

-- Resumen general del sistema
SELECT
    (SELECT COUNT(*) FROM users     WHERE active=1)   AS usuarios_activos,
    (SELECT COUNT(*) FROM catalog_methods WHERE active=1) AS metodos_catalogo,
    (SELECT COUNT(*) FROM work_orders)                AS total_ordenes,
    (SELECT COUNT(*) FROM clients)                    AS total_clientes,
    (SELECT SUM(total_estimated) FROM work_orders
     WHERE status IN ('finalizado','aprobado','en_proceso'))
                                                      AS ingresos_estimados_acumulados;

-- Top 5 métodos más calibrados este año
SELECT
    cm.name  AS metodo,
    cm.area,
    COUNT(oi.id) AS calibraciones
FROM order_instruments oi
JOIN catalog_methods cm ON cm.id = oi.method_id
JOIN work_orders     o  ON o.id  = oi.order_id
WHERE YEAR(o.created_at) = YEAR(NOW())
GROUP BY cm.id
ORDER BY calibraciones DESC
LIMIT 5;

-- Actividad del sistema por usuario (este mes)
SELECT
    u.nombre, u.role,
    COUNT(DISTINCT o.id) AS ordenes_creadas,
    COUNT(DISTINCT d.id) AS documentos_generados
FROM users u
LEFT JOIN work_orders o ON o.created_by = u.id
    AND MONTH(o.created_at) = MONTH(NOW())
    AND YEAR(o.created_at)  = YEAR(NOW())
LEFT JOIN documents d ON d.generated_by = u.id
    AND MONTH(d.generated_at) = MONTH(NOW())
    AND YEAR(d.generated_at)  = YEAR(NOW())
WHERE u.active = 1
GROUP BY u.id
ORDER BY ordenes_creadas DESC;

-- Puntos de calibración más seleccionados
SELECT
    ip.point_label AS punto,
    cm.name        AS metodo,
    COUNT(*)       AS veces_seleccionado
FROM instrument_points ip
JOIN order_instruments oi ON oi.id = ip.instrument_id
JOIN catalog_methods  cm  ON cm.id = oi.method_id
GROUP BY ip.point_label, cm.id
ORDER BY veces_seleccionado DESC
LIMIT 20;
